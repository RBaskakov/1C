	НомерПутевки = ПолучитьЗначениеПоКлючу(ВыбранноеЗначение);
	КолСтрокТЧ = Объект.Путевки.Количество();

		СтрокаТЧ = Объект.Путевки.Получить(НомСтр);	
		Если СтрокаТЧ["НомерПутевки"] = НомерПутевки Тогда
			Сообщить("Выбранная путевка "+Строка(НомерПутевки)+" уже присутствует в документе.");
			Возврат;
		КонецЕсли;
	КонецЦикла;
	
	ТекСтрока = Элементы.Путевки.ТекущиеДанные;	
	ТекСтрока.НомерПутевки = НомерПутевки;

		Возврат Ссылка;
	КонецЕсли;
