Функция ПУПоПлану(НачальнаяДата,КонечнаяДата) Экспорт		
	Если НачальнаяДата > КонечнаяДата Тогда 
		Сообщить("Конечная дата должна быть позже начальной.");
	КонецЕсли;	
	ДнейВГоду = ДеньГода(КонецГода(КонечнаяДата)); 
	Коэф = 0;
	Если Год(НачальнаяДата) < Год(КонечнаяДата) Тогда 
		Коэф = Коэф + (ДеньГода(КонецГода(НачальнаяДата))-ДеньГода(НачальнаяДата))/ДеньГода(КонецГода(НачальнаяДата));	
		ТекГод = Год(НачальнаяДата) + 1;
		Пока ТекГод < Год(КонечнаяДата) Цикл 
			Коэф = Коэф + 1;
			ТекГод = ТекГод +1;
		КонецЦикла;
		Коэф = Коэф + ДеньГода(КонечнаяДата)/ДеньГода(КонецГода(КонечнаяДата)); 		
	Иначе 
		Коэф = (ДеньГода(КонечнаяДата)-ДеньГода(НачальнаяДата))/ДеньГода(КонецГода(НачальнаяДата));	
	КонецЕсли;	
	Коэф = Окр(Коэф * Константы.КоличествоПУвГод.Получить(),1);
	Возврат Коэф;
